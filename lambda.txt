To create a IAM role and to create lambda functions for starting and stoping the instances
---------------------------------------------------------------------------------------------

provider "aws" {
  region = "us-east-1"  # Replace with your desired AWS region
}

# IAM Role for Lambda Function
resource "aws_iam_role" "lambda_role" {
  name               = "lambda-role"
  assume_role_policy = jsonencode({
    Version = "2012-10-17",
    Statement = [{
      Effect    = "Allow",
      Principal = { Service = "lambda.amazonaws.com" },
      Action    = "sts:AssumeRole"
    }]
  })

  tags = {
    Name = "Lambda Execution Role"
  }
}

# IAM Policy for Lambda Function (EC2 Start/Stop Permissions)
resource "aws_iam_policy" "lambda_policy" {
  name        = "lambda-policy"
  description = "IAM policy for Lambda to start and stop EC2 instances"

  policy = jsonencode({
    Version = "2012-10-17",
    Statement = [{
      Effect    = "Allow",
      Action    = [
        "ec2:StartInstances",
        "ec2:StopInstances"
      ],
      Resource  = "arn:aws:ec2:*:*:instance/*"
    }]
  })
}

# Attach IAM Policy to IAM Role
resource "aws_iam_role_policy_attachment" "lambda_policy_attachment" {
  role       = aws_iam_role.lambda_role.name
  policy_arn = aws_iam_policy.lambda_policy.arn
}

# Lambda Function (Start EC2 Instances)
resource "aws_lambda_function" "start_ec2_instances" {
  filename         = "lambda_function.zip"  # Path to your Lambda function code ZIP file
  function_name    = "startEC2Instances"
  role             = aws_iam_role.lambda_role.arn
  handler          = "lambda_function.lambda_handler"
  runtime          = "python3.8"
  timeout          = 60  # Maximum execution time in seconds

  environment {
    variables = {
      OPERATION = "start"
    }
  }

  tags = {
    Name = "StartEC2InstancesFunction"
  }
}

# Lambda Function (Stop EC2 Instances)
resource "aws_lambda_function" "stop_ec2_instances" {
  filename         = "lambda_function.zip"  # Path to your Lambda function code ZIP file
  function_name    = "stopEC2Instances"
  role             = aws_iam_role.lambda_role.arn
  handler          = "lambda_function.lambda_handler"
  runtime          = "python3.8"
  timeout          = 60  # Maximum execution time in seconds

  environment {
    variables = {
      OPERATION = "stop"
    }
  }

  tags = {
    Name = "StopEC2InstancesFunction"
  }
}
